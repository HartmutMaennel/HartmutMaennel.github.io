Filled polygons

from matplotlib.patches import Polygon
from matplotlib.collections import PatchCollection
patches = []
corners = [[5,0],[2,4],[-2,4],[-5,0],[-2,-4],[2,-4],[5,0]]
for i in range(6):
polygon = Polygon([[0,0],corners[i], corners[i+1],[0,0]])
patches.append(polygon)
p = PatchCollection(patches, cmap=matplotlib.cm.jet, alpha=0.4)
p.set_array(np.arange(0,1.01,0.2)) # different colors
axes.add_collection(p)

from mpl_toolkits.mplot3d import axes3d, Axes3D
x = np.linspace(0, nx - 1, nx)
y = np.linspace(0, ny - 1, ny)
x, y = np.meshgrid(x, y)
result = np.sin(np.sqrt(x**2 + y**2))
fig = plt.figure()
ax = Axes3D(fig)
ax.plot_surface(x, y, result, rstride = 1, cstride =1 )
